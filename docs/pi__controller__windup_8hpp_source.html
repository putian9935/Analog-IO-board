21 void update() override {
22 float error =
23 (reader() - reference->get_reference()); // start with an error
24 integral += error;
25 float new_out = kp * error + ki * integral + 32767.;
26 if (new_out > upper ) {
27 new_out = upper;
28 integral -= error;
29 }
30 else if (new_out < lower ) {
31 new_out = lower;
32 integral -= error;
33 }
34 writer((uint16_t)new_out);
35 }
