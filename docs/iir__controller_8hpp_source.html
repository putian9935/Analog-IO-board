21// first order IIR controller with zero and pole
22struct IIRFirstOrderController : public IIRBaseController {
23 float a, b, c;
24 double z, p;
25 IIRFirstOrderController(float const z, float const p)
26 : a((1 + M_PI * p) / (1 - M_PI * p)),
27 b((1 + M_PI * z) / (1 - M_PI * z)),
28 c((1 - M_PI * z) / (1 - M_PI * p)),
29 z(z),
30 p(p) {}
31 IIRFirstOrderController(IIRFirstOrderController const& old) = default;
32 IIRFirstOrderController(IIRFirstOrderController&& old) = default;
33
34 IIRFirstOrderController& operator=(const IIRFirstOrderController& old) {
35 z = old.z;
36 p = old.p;
37 a = old.a;
38 b = old.b;
39 c = old.c;
40 return *this;
41 }
42 IIRFirstOrderController() : z(0), p(0) {}
43 float transfer(float const e) override {
44 float ret = (a * lo - b * le + e);
45 lo = ret;
46 le = e;
47 return ret;
48 }
49 void show() override { Serial.printf("Zero: %.3f; Pole: %.3f\n", z, p); }
50};

53struct IIRSinglePoleController : public IIRBaseController {
54 float a, b;
55 double p;
56 IIRSinglePoleController(double const p)

78template <int len_zeroes, int len_poles>
79struct IIRCascadeController : public Controller {
80 static_assert((len_zeroes == 0 && len_poles == 0) ||
81 (len_zeroes < len_poles),
82 "There should be more poles than zeroes. ");
83
84 float overall_gain;
85 float last_out;
86 float real_g;
87 IIRFirstOrderController focontrollers[len_zeroes];
88 IIRSinglePoleController spcontrollers[len_poles - len_zeroes];
89 IIRCascadeController(read_func_t reader, write_func_t writer,
90 double const (&zeroes)[len_zeroes],
91 double const (&poles)[len_poles],
92 double const overall_gain, double const lower,
93 double const upper, ReferencePath* reference,
94 ReferencePath* reference_hsp)
95 : Controller(reader, writer, reference, reference_hsp, lower, upper),
96 overall_gain(overall_gain),
97 last_out(0.) {
98 int i = 0, j = 0;
99 real_g = overall_gain;
100 for (; i < len_zeroes; ++i) {
101 focontrollers[i] = IIRFirstOrderController(zeroes[i], poles[i]);
102 real_g *= focontrollers[i].a;
103 }
104 for (; i < len_poles; ++i, ++j) {
105 spcontrollers[j] = IIRSinglePoleController(poles[i]);
106 real_g *= spcontrollers[j].a;
107 }
108 }
109
110 void update() override {
145 float new_out =
146 reader() - reference->get_reference(); // start with an error
147 // Serial.printf("%lf\n", reference->get_reference());
148 int i = 0, j = 0;
149 for (; i < len_zeroes; ++i) {
150 new_out = focontrollers[i].transfer(new_out);
151 }
152 for (; i < len_poles; ++i, ++j) {
153 new_out = spcontrollers[j].transfer(new_out);
154 }
155 new_out = max(min(real_g * new_out, upper), lower);
156 if (fabs(new_out - last_out) > 1.f) // lazy update
157 {
158 last_out = new_out;
159 writer((uint16_t)last_out);
160 }
161 }
162
163 void clear() override {
164 int i = 0, j = 0;
165 for (; i < len_zeroes; ++i) focontrollers[i].clear();
166 for (; i < len_poles; ++i, ++j) spcontrollers[j].clear();
167 }
168
169 void read_from_serial(uint8_t const) override;
170};

199template <int a, int b>
200void servo_parser(IIRCascadeController<a, b>* c) {
201 int i = 0, j = 0;
202 c->real_g = 1;
203 for (; i < a; ++i) {
204 double z = SerialReader();
205 double p = SerialReader(); // no guarantee that p is 0
206 c->focontrollers[i] = IIRFirstOrderController(z, p);
207 c->real_g *= c->focontrollers[i].a;
208 }
209 // last controller is always a single pole at -1/2
210 for (; i < (b - 1); ++i, ++j) {
211 double p = SerialReader();
212 c->spcontrollers[j] = IIRSinglePoleController(p);
213 c->real_g *= c->spcontrollers[j].a;
214 }
215 c->real_g *= c->spcontrollers[j].a;
216 // new overall gain
217 c->overall_gain = (double)SerialReader();
218 Serial.printf("overall_gain %f\n", c->overall_gain);
219 c->real_g *= c->overall_gain;
220}
