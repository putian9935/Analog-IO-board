8struct IIRBaseController {
9 double le, lo; // last error and last output
10 IIRBaseController() : le{}, lo{} {}
11 virtual double transfer(double const err) = 0;
12 virtual void show() = 0;
13 virtual ~IIRBaseController(){};

16// first order IIR controller with zero and pole
17struct IIRFirstOrderController : public IIRBaseController {
18 double const z, p;
19 IIRFirstOrderController(double const z, double const p) : z(z), p(p) {}
20 double transfer(double const e) override {
21 double ret =
22 ((1 + M_PI * p) * lo - (1 + M_PI * z) * le + (1 - M_PI * z) * e) /
23 (1 - M_PI * p);
24 lo = ret;
25 le = e;
26 return ret;
27 }
28 void show() override { Serial.printf("Zero: %.3f; Pole: %.3f\n", z, p); }
29};
30
31// first order IIR controller with only pole
32struct IIRSinglePoleController : public IIRBaseController {
33 double const p;
34 IIRSinglePoleController(double const p) : p(p) {}
35 double transfer(double const e) override {
36 double ret = ((1 + M_PI * p) * lo + M_PI * (le + e)) / (1 - M_PI * p);
37 lo = ret;
38 le = e;
39 return ret;
40 }
41 void show() override { Serial.printf("Pole: %.3f\n", p); }
42};

44// IIR controller in cascade form
45template <int len_zeroes, int len_poles>
46struct IIRCascadeController : public Controller {
47 static_assert(len_zeroes < len_poles,
48 "There should be more poles than zeroes. ");
49
50 double overall_gain;
51 double last_out;
52 std::vector<IIRBaseController*> controllers;
53
54 IIRCascadeController(read_func_t reader, write_func_t writer,
55 double const (&zeroes)[len_zeroes],
56 double const (&poles)[len_poles],
57 double const overall_gain, double const lower,
58 double const upper, ReferencePath* reference)
59 : Controller(reader, writer, reference, lower, upper),
60 overall_gain(overall_gain),
61 last_out(0.) {
62 int i = 0;
63 for (; i < len_zeroes; ++i)
64 controllers.push_back(
65 new IIRFirstOrderController(zeroes[i], poles[i]));
66 for (; i < len_poles; ++i)
67 controllers.push_back(new IIRSinglePoleController(poles[i]));
68 }
69
70 void update() override {
71 double new_out =
72 reader() - reference->get_reference(); // start with an error
73 for (auto p : controllers) new_out = p->transfer(new_out);
74 new_out = max(min(overall_gain * new_out, upper), lower);
75 if (abs(new_out - last_out) > 1.) // lazy update
76 {
77 writer(int(new_out));
78 last_out = (int)new_out;
79 }
80 }
81 void read_from_serial(uint8_t const) override;
82};

110template <int a, int b>
111void servo_parser(IIRCascadeController<a, b>* c) {
112 int i = 0;
113 while (i < a) {
114 delete c->controllers[i];
115 double z = SerialReader();
116 double p = SerialReader(); // no guarantee that p is 0
117 c->controllers[i++] = new IIRFirstOrderController(z, p);
118 }
119 // last controller is always a single pole at -1/2
120 while ((i + 1) < b) {
121 delete c->controllers[i];
122 double p = SerialReader();
123 c->controllers[i++] = new IIRSinglePoleController(p);
124 }
125 // new overall gain
126 c->overall_gain = SerialReader();
127 // new waveform
128 c->reference->set_data_from_serial();
129 Serial.printf("Read %d datapoints\n", c->reference->tot);
130 // clear everything
131 c->reference->clear_reference();
132 c->reference->clear_timer();
133 // turn on the controller
134 c->on = true;
135}
